FA      ::= States Alphabet Transitions InitialState FinalStates
States  ::= "States:" State ("," State)*
Alphabet ::= "Alphabet:" Symbol ("," Symbol)*
Transitions ::= "Transitions:" Transition ("," Transition)*
Transition ::= State "->" Symbol "->" State
InitialState ::= "InitialState:" State
FinalStates ::= "FinalStates:" State ("," State)*
State ::= [a-zA-Z][a-zA-Z0-9_]*
Symbol ::= [^,]+
